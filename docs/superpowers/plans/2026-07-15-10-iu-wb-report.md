```typescript
// lib/domain/weeks.ts
// Коды недель ГГНН = (год % 100) * 100 + ISO-неделя (Пн–Вс) — формат
// ориентира БДР и недельных таблиц отчётов.
const DAY_MS = 24 * 60 * 60 * 1000

// ISO-неделя и ISO-год по UTC-полуночи даты.
function isoParts(date: Date): { year: number; week: number } {
  const d = new Date(Date.UTC(date.getUTCFullYear(), date.getUTCMonth(), date.getUTCDate()))
  const dow = d.getUTCDay() || 7 // Пн=1…Вс=7
  d.setUTCDate(d.getUTCDate() + 4 - dow) // четверг определяет ISO-год
  const year = d.getUTCFullYear()
  const jan1 = new Date(Date.UTC(year, 0, 1))
  const week = Math.ceil(((d.getTime() - jan1.getTime()) / DAY_MS + 1) / 7)
  return { year, week }
}

export function weekCode(date: Date): number {
  const { year, week } = isoParts(date)
  return (year % 100) * 100 + week
}

// Понедельник недели с кодом ГГНН (UTC-полночь). Годы трактуем как 20ГГ.
export function weekStart(code: number): Date {
  const year = 2000 + Math.floor(code / 100)
  const week = code % 100
  const jan4 = new Date(Date.UTC(year, 0, 4)) // 4 января всегда в W01
  const dow = jan4.getUTCDay() || 7
  const mondayW1 = new Date(jan4.getTime() - (dow - 1) * DAY_MS)
  return new Date(mondayW1.getTime() + (week - 1) * 7 * DAY_MS)
}

export function weekEnd(code: number): Date {
  return new Date(weekStart(code).getTime() + 6 * DAY_MS)
}

export function compareWeeks(code: number): {
  prev: number
  monthAgo: number
  yearAgo: number
} {
  const start = weekStart(code)
  const shift = (weeks: number) =>
    weekCode(new Date(start.getTime() - weeks * 7 * DAY_MS))
  return { prev: shift(1), monthAgo: shift(4), yearAgo: shift(52) }
}
```

```typescript
// lib/domain/iu.ts
// Метрики отчёта «ИУ WB план-факт». Чистая логика (рубли number — отчётные
// агрегаты). Формулы: fin/finflow/02-iu-report.md.
export type IuDayRow = {
  date: string // YYYY-MM-DD
  buyoutRub: number
  ordersRub: number
  advRub: number
  buyoutQty: number
  ordersQty: number
}

export type IuSettings = {
  bannerTotal: number
  bannerFrom: string | null
  bannerTo: string | null
  advExtraAgreement: number
  advPercent: number
  targetBuyoutH1: number
  targetBuyoutH2: number
  planBuyoutDailyRef: number
}

export const IU_SETTING_KEYS = [
  "bannerTotal",
  "advExtraAgreement",
  "advPercent",
  "targetBuyoutH1",
  "targetBuyoutH2",
  "planBuyoutDailyRef",
] as const

// Баннер ВБ: в API-статистике его нет — добавляем равномерной долей
// в дни периода действия (из настроек).
export function applyBanner(days: IuDayRow[], s: IuSettings): IuDayRow[] {
  if (!s.bannerFrom || !s.bannerTo || s.bannerTotal <= 0) return days
  const from = s.bannerFrom
  const to = s.bannerTo
  const periodDays =
    Math.round(
      (new Date(to).getTime() - new Date(from).getTime()) / 86_400_000
    ) + 1
  if (periodDays <= 0) return days
  const share = s.bannerTotal / periodDays
  return days.map((d) =>
    d.date >= from && d.date <= to ? { ...d, advRub: d.advRub + share } : d
  )
}

// KPI «Выкупы вчера»: план = требуемый темп до цели полугодия
// (НЕ «План WB» и не «ПЛАН на день» — решение из доки миграции).
export function needDaily(input: {
  target: number
  factToDate: number
  daysLeft: number
}): number | null {
  if (input.daysLeft <= 0) return null
  return (input.target - input.factToDate) / input.daysLeft
}

export function percentDone(fact: number, target: number): number | null {
  if (target === 0) return null
  return (fact / target) * 100
}

// ДРР = траты / база. База различает два ДРР отчёта:
// «Заказы·ДРР» — ordersRub; обязательство — buyoutRub.
export function drr(
  days: IuDayRow[],
  base: "buyoutRub" | "ordersRub"
): number | null {
  let spent = 0
  let denom = 0
  for (const d of days) {
    spent += d.advRub
    denom += d[base]
  }
  if (denom === 0) return null
  return spent / denom
}

// Прогноз трат = потрачено + недельный ДРР (последние 7 завершённых дней,
// по выкупам) × остаток выкупов до цели.
export function advForecast(input: {
  spent: number
  weeklyDrr: number | null
  buyoutLeft: number
}): number | null {
  if (input.weeklyDrr === null) return null
  return input.spent + input.weeklyDrr * input.buyoutLeft
}

// Обязательство по рекламе: план трат (Σ planAdv периода, ≈ advPercent%
// плановой выручки) + допсоглашение — только для П1.
export function obligation(input: {
  planAdvSum: number
  half: 0 | 1 | 2 // 0 = весь год
  settings: IuSettings
}): number {
  const extra = input.half === 2 ? 0 : input.settings.advExtraAgreement
  return input.planAdvSum + extra
}

export function halfOf(dateStr: string): 1 | 2 {
  return Number(dateStr.slice(5, 7)) <= 6 ? 1 : 2
}
```

```typescript
const METRICS: IuMetric[] = [
  "buyout_rub",
  "orders_rub",
  "adv_rub",
  "buyout_qty",
  "orders_qty",

type CacheDay = Record<string, number | boolean | string | null>

export const fixtureIuFactFetcher: IuFactFetcher = {
  async fetch(dateFrom, dateTo) {
```

```typescript
// lib/sync/run-reports-sync.ts
// Синк scope=reports: тяжёлые отчётные выборки, отдельное расписание
// (ночной cron + кнопка на отчёте). Шаги независимы; окно перезаписи —
// 35 дней (корректировки дотекают задним числом), full — вся история с 1 января.
import { prisma } from "@/lib/db"
import { getIuFactFetcher } from "@/lib/integrations/iu-facts"
import type { Prisma, SyncTrigger } from "@prisma/client"

const WINDOW_DAYS = 35
const RUNNING_STALE_MS = 30 * 60 * 1000

export type ReportsSyncResult =
  | { skipped: true }
  | { skipped: false; runId: string; status: "ok" | "error" }

type StepReport = Record<string, { upserted: number } | { error: string }>

async function step(
  fn: () => Promise<number>
): Promise<{ upserted: number } | { error: string }> {
  try {
    return { upserted: await fn() }
  } catch (e) {
    return { error: e instanceof Error ? e.message : String(e) }
  }
}

function isoDate(d: Date): string {
  return d.toISOString().slice(0, 10)
}

async function syncIuFacts(dateFrom: string, dateTo: string): Promise<number> {
  const rows = await getIuFactFetcher().fetch(dateFrom, dateTo)
  const syncedAt = new Date()
  for (const r of rows) {
    const data = {
      metric: r.metric,
      accountId: r.accountId,
      date: new Date(r.date),
      value: r.value,
      syncedAt,
    }
    await prisma.iuFactDaily.upsert({
      where: {
        metric_accountId_date: {
          metric: r.metric,
          accountId: r.accountId,
          date: new Date(r.date),
        },
      },
      update: data,
      create: data,
    })
  }
  return rows.length
}

export async function runReportsSync(
  trigger: SyncTrigger,
  options: { full?: boolean } = {}
): Promise<ReportsSyncResult> {
  const running = await prisma.syncRun.findFirst({
    where: {
      scope: "reports",
      status: "running",
      startedAt: { gt: new Date(Date.now() - RUNNING_STALE_MS) },
    },
  })
  if (running) return { skipped: true }

  const run = await prisma.syncRun.create({
    data: { status: "running", trigger, scope: "reports" },
  })

  const now = new Date()
  const from = options.full
    ? `${now.getUTCFullYear()}-01-01`
    : isoDate(new Date(now.getTime() - WINDOW_DAYS * 86_400_000))
  const to = isoDate(now)

  const report: StepReport = {}
  report.iu_facts = await step(() => syncIuFacts(from, to))
  // План 11 добавит шаги bdr_wb / bdr_ozon / bdr_cogs сюда же.

  const hasError = Object.values(report).some((r) => "error" in r)
  await prisma.syncRun.update({
    where: { id: run.id },
    data: {
      status: hasError ? "error" : "ok",
      finishedAt: new Date(),
      slices: report as Prisma.InputJsonValue,
      error: hasError
        ? Object.entries(report)
            .filter(([, r]) => "error" in r)
            .map(([k, r]) => `${k}: ${(r as { error: string }).error}`)
            .join("; ")
        : null,
    },
  })
  return { skipped: false, runId: run.id, status: hasError ? "error" : "ok" }
}
```

```typescript
// lib/reports/iu-plan-parser.ts
// Парсер листа «2026 ПЛАН» файла плана ИУ (порт fin/finflow-app/plan.py):
// заголовки — строка 2, данные с 3-й; колонки ищем по заголовку.
import ExcelJS from "exceljs"

export type IuPlanParsedRow = {
  date: string // YYYY-MM-DD
  planBuyout: number
  planWb: number
  planOrders: number
  planAdv: number
}

const SHEET = "2026 ПЛАН"
const HEADER_ROW = 2

const COLS: Record<string, keyof Omit<IuPlanParsedRow, "date">> = {
  "ПЛАН на день": "planBuyout",
  "План WB  учитывая нереализованное в рублях в выкупах до вычета спп":
    "planWb",
  "Новый план заказы в руб": "planOrders",
  "Траты рекламные план от плана": "planAdv",
}

function cellDate(v: ExcelJS.CellValue): string | null {
  if (v instanceof Date) return v.toISOString().slice(0, 10)
  if (typeof v === "string" && /^\d{4}-\d{2}-\d{2}/.test(v)) return v.slice(0, 10)
  return null
}

function cellNumber(v: ExcelJS.CellValue): number {
  if (typeof v === "number" && Number.isFinite(v)) return v
  if (v && typeof v === "object" && "result" in v && typeof v.result === "number")
    return v.result // формульная ячейка: exceljs отдаёт вычисленное значение
  return 0
}

export async function parseIuPlan(buffer: Buffer): Promise<IuPlanParsedRow[]> {
  const wb = new ExcelJS.Workbook()
  await wb.xlsx.load(buffer)
  const ws = wb.getWorksheet(SHEET)
  if (!ws) throw new Error(`В файле нет листа «${SHEET}»`)

  // Колонка → индекс по заголовкам строки 2.
  const colIndex = new Map<string, number>()
  ws.getRow(HEADER_ROW).eachCell((cell, col) => {
    const text = String(cell.value ?? "").trim()
    if (text) colIndex.set(text, col)
  })
  const required = ["Дата", ...Object.keys(COLS)]
  const missing = required.filter((c) => !colIndex.has(c))
  if (missing.length > 0)
    throw new Error(`В листе «${SHEET}» нет колонок: ${missing.join(", ")}`)

  const rows: IuPlanParsedRow[] = []
  for (let r = HEADER_ROW + 1; r <= ws.rowCount; r++) {
    const row = ws.getRow(r)
    const date = cellDate(row.getCell(colIndex.get("Дата")!).value)
    if (!date) continue
    const parsed: IuPlanParsedRow = {
      date,
      planBuyout: 0,
      planWb: 0,
      planOrders: 0,
      planAdv: 0,
    }
    for (const [header, key] of Object.entries(COLS)) {
      parsed[key] = cellNumber(row.getCell(colIndex.get(header)!).value)
    }
    rows.push(parsed)
  }
  return rows
}
```

```typescript
// app/reports/iu-wb/settings/actions.ts
"use server"

import { revalidatePath } from "next/cache"
import { prisma } from "@/lib/db"
import { requireAction } from "@/lib/auth/session"
import { IU_SETTING_KEYS } from "@/lib/domain/iu"
import { parseIuPlan } from "@/lib/reports/iu-plan-parser"

export type FormState = { error: string | null; ok?: string }

const MAX_XLSX_BYTES = 20 * 1024 * 1024

export async function uploadIuPlan(
  _prevState: FormState,
  formData: FormData
): Promise<FormState> {
  const auth = await requireAction("manage_report_settings")
  if (auth.error) return { error: auth.error }

  const file = formData.get("file")
  if (!(file instanceof File) || file.size === 0)
    return { error: "Выберите xlsx-файл плана" }
  if (file.size > MAX_XLSX_BYTES) return { error: "Файл больше 20 МБ" }

  let rows
  try {
    rows = await parseIuPlan(Buffer.from(await file.arrayBuffer()))
  } catch (e) {
    return { error: e instanceof Error ? e.message : String(e) }
  }
  if (rows.length === 0) return { error: "В файле не нашлось план-строк с датами" }

  // Активен последний аплоад: старые план-строки замещаются целиком.
  await prisma.$transaction(async (tx) => {
    const upload = await tx.iuPlanUpload.create({
      data: {
        fileName: file.name,
        rowsParsed: rows.length,
        uploadedById: auth.user.id,
        uploadedBy: auth.user.name,
      },
    })
    await tx.iuPlanDay.deleteMany({})
    await tx.iuPlanDay.createMany({
      data: rows.map((r) => ({
        date: new Date(r.date),
        planBuyout: r.planBuyout,
        planWb: r.planWb,
        planOrders: r.planOrders,
        planAdv: r.planAdv,
        uploadId: upload.id,
      })),
    })
  })

  revalidatePath("/reports/iu-wb")
  revalidatePath("/reports/iu-wb/settings")
  return { error: null, ok: `Загружено строк: ${rows.length}` }
}

export async function saveIuSettings(
  _prevState: FormState,
  formData: FormData
): Promise<FormState> {
  const auth = await requireAction("manage_report_settings")
  if (auth.error) return { error: auth.error }

  for (const key of IU_SETTING_KEYS) {
    const raw = String(formData.get(key) ?? "").replace(/\s/g, "").replace(",", ".")
    const value = Number(raw)
    if (!Number.isFinite(value) || value < 0)
      return { error: `Параметр «${key}» должен быть неотрицательным числом` }
    const dateFrom = String(formData.get(`${key}_from`) ?? "")
    const dateTo = String(formData.get(`${key}_to`) ?? "")
    await prisma.iuReportSetting.upsert({
      where: { key },
      update: {
        value,
        dateFrom: dateFrom ? new Date(dateFrom) : null,
        dateTo: dateTo ? new Date(dateTo) : null,
      },
      create: {
        key,
        value,
        dateFrom: dateFrom ? new Date(dateFrom) : null,
        dateTo: dateTo ? new Date(dateTo) : null,
      },
    })
  }
  revalidatePath("/reports/iu-wb")
  revalidatePath("/reports/iu-wb/settings")
  return { error: null, ok: "Сохранено" }
}
```

```tsx
export default async function IuReportPage({
  searchParams,

  searchParams: Promise<Search>

  const sp = await searchParams
  const half = Number(typeof sp.half === "string" ? sp.half : "0") as 0 | 1 | 2

  const user = await getCurrentUser()
  const canSettings = user && can(user.role as Role, "manage_report_settings")

  const [factRows, planRows, settingRows, lastSync] = await Promise.all([
    prisma.iuFactDaily.findMany({ orderBy: { date: "asc" } }),
    prisma.iuPlanDay.findMany({ orderBy: { date: "asc" } }),
    prisma.iuReportSetting.findMany(),
    prisma.syncRun.findFirst({
      where: { scope: "reports", status: { in: ["ok", "error"] } },
      orderBy: { startedAt: "desc" },
    }),
  ])

  const s = new Map(settingRows.map((r) => [r.key, r]))
  const num = (key: string) => Number(s.get(key)?.value ?? 0)
  const settings: IuSettings = {
    bannerTotal: num("bannerTotal"),
    bannerFrom: s.get("bannerTotal")?.dateFrom?.toISOString().slice(0, 10) ?? null,
    bannerTo: s.get("bannerTotal")?.dateTo?.toISOString().slice(0, 10) ?? null,
    advExtraAgreement: num("advExtraAgreement"),
    advPercent: num("advPercent"),
    targetBuyoutH1: num("targetBuyoutH1"),
    targetBuyoutH2: num("targetBuyoutH2"),
    planBuyoutDailyRef: num("planBuyoutDailyRef"),
  }

  // Дневные строки: метрики в колонки (accountId=0 — сумма обоих кабинетов).
  const byDate = new Map<string, IuDayRow>()
  for (const r of factRows) {
    const date = r.date.toISOString().slice(0, 10)
    const row =
      byDate.get(date) ??
      ({ date, buyoutRub: 0, ordersRub: 0, advRub: 0, buyoutQty: 0, ordersQty: 0 } as IuDayRow)
    if (r.metric === "buyout_rub") row.buyoutRub += Number(r.value)
    if (r.metric === "orders_rub") row.ordersRub += Number(r.value)
    if (r.metric === "adv_rub") row.advRub += Number(r.value)
    if (r.metric === "buyout_qty") row.buyoutQty += Number(r.value)
    if (r.metric === "orders_qty") row.ordersQty += Number(r.value)
    byDate.set(date, row)
  }
  const allDays = applyBanner(
    [...byDate.values()].sort((a, b) => a.date.localeCompare(b.date)),
    settings
  )

  const inHalf = (date: string) => half === 0 || halfOf(date) === half
  const days = allDays.filter((d) => inHalf(d.date))
  const plans = planRows
    .map((p) => ({
      date: p.date.toISOString().slice(0, 10),
      planBuyout: Number(p.planBuyout),
      planWb: Number(p.planWb),
      planOrders: Number(p.planOrders),
      planAdv: Number(p.planAdv),
    }))
    .filter((p) => inHalf(p.date))

  // KPI: «вчера» = последний день с фактом.
  const yesterday = days.at(-1) ?? null
  const target =
    half === 2
      ? settings.targetBuyoutH2
      : half === 1
        ? settings.targetBuyoutH1
        : settings.targetBuyoutH1 + settings.targetBuyoutH2
  const factToDate = days.reduce((sum, d) => sum + d.buyoutRub, 0)
  const halfEnd = half === 1 ? "-06-30" : "-12-31"
  const year = (yesterday?.date ?? "2026-01-01").slice(0, 4)
  const daysLeft = yesterday
    ? Math.max(
        0,
        Math.round(
          (new Date(`${year}${halfEnd}`).getTime() - new Date(yesterday.date).getTime()) /
            86_400_000
        )
      )
    : 0
  const need = needDaily({ target, factToDate, daysLeft })
  const pct = percentDone(factToDate, target)
  const spent = days.reduce((sum, d) => sum + d.advRub, 0)
  const weeklyDrr = drr(days.slice(-8, -1), "buyoutRub") // 7 завершённых дней
  const forecast = advForecast({
    spent,
    weeklyDrr,
    buyoutLeft: Math.max(0, target - factToDate),
  })
  const oblig = obligation({
    planAdvSum: plans.reduce((sum, p) => sum + p.planAdv, 0),
    half,
    settings,
  })

  // Недельная агрегация для графиков и таблицы.
  const weeks = new Map<
    number,
    { plan: number; fact: number; adv: number; orders: number; planAdv: number }
  >()
  const wk = (code: number) =>
    weeks.get(code) ?? { plan: 0, fact: 0, adv: 0, orders: 0, planAdv: 0 }
  for (const d of days) {
    const code = weekCode(new Date(d.date))
    const w = wk(code)
    w.fact += d.buyoutRub
    w.adv += d.advRub
    w.orders += d.ordersRub
    weeks.set(code, w)
  }
  for (const p of plans) {
    const code = weekCode(new Date(p.date))
    const w = wk(code)
    w.plan += p.planBuyout
    w.planAdv += p.planAdv
    weeks.set(code, w)
  }
  const codes = [...weeks.keys()].sort((a, b) => a - b)
  const weekly: WeekPoint[] = []
  const cum: CumPoint[] = []
  const adv: AdvPoint[] = []
  const drrPoints: DrrPoint[] = []
  let planCum = 0
  let factCum = 0
  let spentCum = 0
  let obligCum = 0
  for (const code of codes) {

    const label = `W${code % 100}`
    planCum += w.plan
    factCum += w.fact
    spentCum += w.adv
    obligCum += w.planAdv
    weekly.push({ week: label, plan: Math.round(w.plan), fact: Math.round(w.fact) })
    cum.push({ week: label, planCum: Math.round(planCum), factCum: Math.round(factCum) })
    adv.push({
      week: label,
      spentCum: Math.round(spentCum),
      obligation: Math.round(obligCum + (half === 2 ? 0 : settings.advExtraAgreement)),
    })
    drrPoints.push({
      week: label,
      ordersRub: Math.round(w.orders),
      drrPct: w.orders > 0 ? Math.round((w.adv / w.orders) * 1000) / 10 : 0,
    })
  }

  const kpi = [
    {
      label: "Выкупы вчера",

    },
    {
      label: "Реклама накопительно",
      value: fmtRub(spent),
      sub: `обязательство ${fmtRub(oblig)}`,
    },
    {
      label: "Прогноз трат",

    },
  ]

  return (
    <main className="mx-auto max-w-6xl space-y-6 p-8">
      <div className="flex flex-wrap items-center justify-between gap-4">
        <h1 className="text-2xl font-semibold">ИУ WB план-факт</h1>

                timeStyle: "short",
              })}
            </span>
          ) : (
            <span>Данные ещё не загружались</span>

            <Link href="/reports/iu-wb/settings" className="underline underline-offset-4">

            </Link>
          )}
        </div>
      </div>

      <div className="flex flex-wrap items-center gap-2">
```

```typescript
// Выручка/шт выкупов — КУБ (v_wb_sales_complete): источник истины,
// реконструкция из wb_sales расходилась на горизонте года (~0,5%).
const SQL_BUYOUT = `
SELECT CAST([Дата продажи] AS DATE) d,
       ROUND(SUM([Цена со скидкой]), 2) buyout_rub,
       SUM(CASE WHEN LEFT([Продажа ID],1)='S' THEN 1 ELSE -1 END) buyout_qty
FROM WBALL.v_wb_sales_complete
WHERE CAST([Дата продажи] AS DATE) BETWEEN @from AND @to
  AND [МП Аккаунт ID] IN (1,2)
GROUP BY CAST([Дата продажи] AS DATE)`

// Заказы — базовая wb_orders со srid-дедупом (вьюха заказов битая:
// теряет март/февраль — см. доку миграции).
const SQL_ORDERS = `
WITH d AS (
  SELECT CAST(order_dt AS DATE) d, price_with_disc,
         ROW_NUMBER() OVER (PARTITION BY srid ORDER BY batch_id DESC) rn
  FROM WBALL.wb_orders
  WHERE CAST(order_dt AS DATE) BETWEEN @from AND @to AND account_id IN (1,2)
)
SELECT d, ROUND(SUM(price_with_disc), 2) orders_rub, COUNT(*) orders_qty
FROM d WHERE rn = 1 GROUP BY d`

// Реклама — fullstats с дедупом по (account, campaign, день).
// Кластерную рекламу НЕ добавляем (двойной счёт); баннер — домен (настройки).
const SQL_ADV = `
WITH r AS (
  SELECT CAST(report_date AS DATE) d, spend,
         ROW_NUMBER() OVER (
           PARTITION BY account_id, campaign_id, CAST(report_date AS DATE)
           ORDER BY batch_id DESC) rn
  FROM WBALL.wb_adv_fullstats_daily
  WHERE CAST(report_date AS DATE) BETWEEN @from AND @to AND account_id IN (1,2)
)
SELECT d, ROUND(SUM(spend), 2) adv_rub FROM r WHERE rn = 1 GROUP BY d`

async function query(
  text: string,
  from: string,
  to: string
): Promise<Record<string, unknown>[]> {
  const pool = await getMssqlPool()
  const request = pool.request()
  request.input("from", sql.Date, from)
  request.input("to", sql.Date, to)
  return (await request.query(text)).recordset
}

function push(
  rows: IuFactRow[],
  metric: IuFactRow["metric"],
  raw: Record<string, unknown>[],
  col: string
) {
  for (const r of raw) {
    const value = Number(r[col])
    if (!Number.isFinite(value)) continue
    rows.push({
      metric,
      accountId: 0,
      date: (r.d as Date).toISOString().slice(0, 10),
      value,
    })
  }
}

export const mssqlIuFactFetcher: IuFactFetcher = {
  async fetch(dateFrom, dateTo) {
    const [buyout, orders, adv] = await Promise.all([
      query(SQL_BUYOUT, dateFrom, dateTo),
      query(SQL_ORDERS, dateFrom, dateTo),
      query(SQL_ADV, dateFrom, dateTo),
    ])
    const rows: IuFactRow[] = []
    push(rows, "buyout_rub", buyout, "buyout_rub")
    push(rows, "buyout_qty", buyout, "buyout_qty")
    push(rows, "orders_rub", orders, "orders_rub")
    push(rows, "orders_qty", orders, "orders_qty")
    push(rows, "adv_rub", adv, "adv_rub")
    return rows
  },
}
```
